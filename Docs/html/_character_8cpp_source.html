132 void Character::attack(Character * enemy) {

135  (*Debug::debugOutput) << ss.rdbuf() ;
136  ss.clear() ;
137  attack_helper(enemy) ;

139  ss << this->name << " wins!" << '\n' ;

157 void Character::attack_helper(Character * enemy) {
158  this->state = CharacterState::inCombat ;
159  if (enemy->state != CharacterState::inCombat) {
160  enemy->attack_helper(this) ;

166  return ;
