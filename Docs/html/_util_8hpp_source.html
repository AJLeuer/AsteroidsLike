169  * Rounds a float or double to an int or long
170  */
171 template<typename F = float, typename I = int> // F = some float or double, I = some unsigned, int, long, etc
172 I roundF(F value) {
173  if ((typeid(F) == typeid(unsigned)) || (typeid(F) == typeid(int)) || (typeid(F) == typeid(long))) {
174  return value ;
175  }
176  else {
177  F temp = (value >= 0.0f) ? (floor(value + 0.5f)) : (ceil(value - 0.5f)) ;
178  I round = static_cast<I>(temp) ;
179 
180  /* catch a weird problem where this was returning a negative value from a posititive input */
181  if ((value >= 0) && (round < 0)) {
182  round = (round * -1) ;
183  }
184 
185  return round ;
186  }
187 }
