53 /**
54  * Translates the Position coords from a source coordinate system to coordinates
55  * in the destination coordinate system. To work properly, this function an additonal
56  * pair of coordinates: first coordinates from the source coordinate system, followed by
57  * equivalent coordinates from the destination coordinate system. "Equivalent" implies that if these two
58  * coordinate systems were overlaid one on top of the other, then these two coordinates would be
59  * in the same spot. The simplest way to do this is to simply give the origin ({0, 0, 0}) from the source
60  * coordinate system, and its equivalent from the destination coordinate system.
61  *
62  * @param coords The coordinate to translate
63  * @param sourceCoords Sample coordinates from the source coordinate system
64  * @param equivalentDestinationCoords Coordinates from the destination coordinate system that are equivalent to sourceCoords
65  */
66 template<typename N>
67 Position<N> translateCoords(const Position<N> & coords, const Position<N> & sourceCoords, const Position<N> & equivalentDestinationCoords) {
68 
69  /* using a position to store, not an actual position, but the difference between a position from the source coordinate system and
70  a position from the destination coordinate system (i.e. a vector) */
71  auto differential = sourceCoords - equivalentDestinationCoords ;
72 
73  auto coordsAsDestCoordSystemCoords = (coords + differential) ;
74 
75  return coordsAsDestCoordSystemCoords ;
76 }

78 /**
79  * An implementation of the formula for finding the distance between two points
80  * in Euclidian space, as given at http://en.wikipedia.org/wiki/Euclidean_distance
81  *
82  * @param p The first point
83  * @param q The second point
84  */
85 template<typename N>
86 N calcEuclidianDistance(const Position<N> & p, const Position<N> & q) {
87  auto sum = pow<N>((q.getX() - p.getX()), 2) + pow<N>((q.getY() - p.getY()), 2) + pow<N>((q.getZ() - p.getZ()), 2) ;
88  auto sqRoot = sqrt(sum) ;
89  return sqRoot ;
90 }
