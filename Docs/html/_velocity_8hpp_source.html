192 
193  if ((velocityStorage.at(i) != nullptr) && ((velocityStorage.at(i)->lastDistance != *velocityStorage.at(i)->distance) || (velocityStorage.at(i)->lastVelocity != 0))) {
194 
195  const N dist0 = *velocityStorage.at(i)->distance ;
196  const auto time0 = velocityStorage.at(i)->timer->checkTimeElapsed() ;
197 
198  this_thread::sleep_for(velocityStorage.at(i)->baseTimeUnit) ;
199 
200  const N dist1 = *velocityStorage.at(i)->distance ;
201  auto time1 = velocityStorage.at(i)->timer->checkTimeElapsed() ;
202 
203  const N totalDistance = dist1 - dist0 ;
204 
205  auto totalTime = chrono::duration_cast<chrono::milliseconds>(time1 - time0) ;
206 
207  auto timePassed = totalTime.count() ;
208 
209  double velocity = (totalDistance / timePassed) ;
210 
211  velocityStorage.at(i)->lastVelocity = velocity ;
212  velocityStorage.at(i)->lastDistance = *velocityStorage.at(i)->distance ;
213 
214  }
215  else {
216  this_thread::sleep_for(defaultSleepTime) ;
217  }
218  }
219  else {
220  this_thread::sleep_for(defaultSleepTime) ;
221  }
222  sharedVelMutex->unlock() ;
