411 bool OutputData<POSUTYPE, SIZEUTYPE>::checkIfUpdated() {

413  /* if the updateFlag was set directly, this overrides any checking. Just return true immediately */
414  for (auto i = 0 ; i < updateFlags.size(); i++) {
415  if (*updateFlags.at(i) == true) {
